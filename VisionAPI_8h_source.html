23 typedef struct ParticleAnalysisReport_struct {
24  int imageHeight;
25  int imageWidth;
26  double imageTimestamp;
27  int particleIndex; // the particle index analyzed
28  /* X-coordinate of the point representing the average position of the
29  * total particle mass, assuming every point in the particle has a constant density */
30  int center_mass_x; // MeasurementType: IMAQ_MT_CENTER_OF_MASS_X
31  /* Y-coordinate of the point representing the average position of the
32  * total particle mass, assuming every point in the particle has a constant density */
33  int center_mass_y; // MeasurementType: IMAQ_MT_CENTER_OF_MASS_Y
34  double center_mass_x_normalized; //Center of mass x value normalized to -1.0 to +1.0 range
35  double center_mass_y_normalized; //Center of mass y value normalized to -1.0 to +1.0 range
36  /* Area of the particle */
37  double particleArea; // MeasurementType: IMAQ_MT_AREA
38  /* Bounding Rectangle */
39  Rect boundingRect; // left/top/width/height
40  /* Percentage of the particle Area covering the Image Area. */
41  double particleToImagePercent; // MeasurementType: IMAQ_MT_AREA_BY_IMAGE_AREA
42  /* Percentage of the particle Area in relation to its Particle and Holes Area */
43  double particleQuality; // MeasurementType: IMAQ_MT_AREA_BY_PARTICLE_AND_HOLES_AREA
44 } ParticleAnalysisReport;
